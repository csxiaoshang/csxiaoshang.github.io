    ```
    public void solution(int[] A,int size,int[] B){
        if(B==null || B.length == 0){
            return;
        }
        int n = size;
        int m = B.length;
        int index = n+m-1;
        for(int i = index;i>=0;i--){
            if(A[n-1]>B[m-1]){
                A[index] = A[n-1];
                n--;
            }else{
                A[index] = B[m-1];
                m--;
            }
            if(m<=0||n<=0){
                break;
            }
        }
        if(n<=0){
            for(int i = index;i>=0;i--){
                A[i] = B[m--]
            }
        }
    }
```
